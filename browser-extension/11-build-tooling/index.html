жһ„е»әдёҺе·ҘзЁӢеҢ–¶

еӨҡе…ҘеҸЈжү“еҢ…¶

жү©еұ•дёҺе…¶д»– Web йЎ№зӣ®зҡ„ж №жң¬еҢәеҲ«пјҡжңүеӨҡдёӘзӢ¬з«Ӣе…ҘеҸЈгҖӮжҜҸдёӘз»„д»¶йғҪжҳҜдёҖеҘ—зӢ¬з«Ӣзҡ„ HTML/CSS/JSгҖӮ

SW е…ҘеҸЈ          вҶ’ background.js
Popup е…ҘеҸЈ       вҶ’ popup.html + popup.js
Content Script   вҶ’ content.jsпјҲзӢ¬з«Ӣ bundleпјү
Side Panel       вҶ’ sidepanel.html + sidepanel.js
Options          вҶ’ options.html + options.js
DevTools         вҶ’ devtools.html + devtools.js

жҜҸдёӘе…ҘеҸЈд№Ӣй—ҙдёҚиғҪ import еҜ№ж–№зҡ„жЁЎеқ—вҖ”вҖ”е®ғд»¬иҝҗиЎҢеңЁдёҚеҗҢзҡ„дёҠдёӢж–ҮдёӯгҖӮ

жү“еҢ…е·Ҙе…·йҖүеһӢ¶

Vite¶

йҖӮеҗҲжү©еұ•ејҖеҸ‘пјҢдҪҶеҺҹз”ҹдёҚж”ҜжҢҒеӨҡе…ҘеҸЈ HTMLпјҡ

// vite.config.js
import { defineConfig } from 'vite';
import { resolve } from 'path';

export default defineConfig({
  build: {
    rollupOptions: {
      input: {
        background: resolve(__dirname, 'src/background/index.ts'),
        popup: resolve(__dirname, 'src/popup/index.html'),
        content: resolve(__dirname, 'src/content/index.ts'),
        sidepanel: resolve(__dirname, 'src/sidepanel/index.html'),
      },
      output: {
        entryFileNames: '[name].js',
      }
    },
    outDir: 'dist'
  }
});

жіЁж„ҸпјҡVite й»ҳи®ӨеҜ№ HTML е…ҘеҸЈдјҡеӨ„зҗҶиө„жәҗи·Ҝеҫ„гҖӮContent Script е’Ң Service Worker жҳҜзәҜ JSпјҢйңҖиҰҒз”Ё build.rollupOptions.input зӣҙжҺҘжҢҮе®ҡгҖӮ

Webpack¶

жҲҗзҶҹж–№жЎҲпјҢзӨҫеҢәжҸ’д»¶дё°еҜҢгҖӮ

// webpack.config.js
const CopyPlugin = require('copy-webpack-plugin');

module.exports = {
  entry: {
    background: './src/background/index.ts',
    popup: './src/popup/index.tsx',
    content: './src/content/index.ts',
  },
  output: {
    path: path.resolve(__dirname, 'dist'),
    filename: '[name].js'
  },
  plugins: [
    new CopyPlugin({
      patterns: [
        { from: 'public/manifest.json', to: 'manifest.json' },
      ],
    }),
  ]
};
